class Node:
    
    def __init__(self, data):

        self.data = data
        self.next = None

class linkedList:

    def __init__(self, data):

        self.head = Node(data)

    def append(self, data):

        cur = self.head

        while cur.next is not None:
            cur = cur.next

        cur.next = Node(data)

    def print_all(self):
        cur = self.head
        while cur is not None:
            print(cur.data)
            cur = cur.next

    def get_node(self, index):

        cur = self.head

        cur_index = 0
        while cur is not None:
            if cur_index == index:
                return cur
            else:
                cur = cur.next

    def add_node(self, index, value):

        new = Node(value)
        
        if index == 0:
            new.next = self.head.next
            self.head = new
            return

        prev = self.get_node(index - 1)

        new.next = prev.next
        prev.next = new

    def delete_node(self, index):

        if index == 0:
            self.head = self.head.next
            return

        prev = self.get_node(index - 1)
        prev.next = prev.next.next
